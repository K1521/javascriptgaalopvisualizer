// Join tests: 2, 3, and 4 points
// Uses DCGA join: Dual(Dual(A) ^ Dual(B) ...)

/* Parameters (sliders)
x1 = 0; y1 = 0; z1 = 0;
x2 = 1; y2 = 0; z2 = 0;
x3 = 0; y3 = 1; z3 = 0;
x4 = 0; y4 = 0; z4 = 1;
*/

// Points
?P1 = EV(x1,y1,z1);
?P2 = EV(x2,y2,z2);
?P3 = EV(x3,y3,z3);
?P4 = EV(x4,y4,z4);

// Join of 2 points → line
?Join2 = Dual(Dual(P1) ^ Dual(P2));

// Join of 3 points → plane
?Join3 = Dual(Dual(P1) ^ Dual(P2) ^ Dual(P3));

// Join of 4 points → tetrahedron blade
?Join4 = Dual(Dual(P1) ^ Dual(P2) ^ Dual(P3) ^ Dual(P4));

// Visualization
:Blue;
:P1;

:Red;
:P2;

:Yellow;
:P3;

:Cyan;
:P4;

:Green;
:Join2;

:Magenta;
:Join3;

:Orange;
:Join4;
