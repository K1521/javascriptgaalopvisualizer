// Join of 4 movable points
// Each point has parameters (x,y,z) from -1 to 1
// Tests 4-point volume, general 3-blade join

/* Parameters (sliders)
x1 = 0; y1 = 0; z1 = 0;
x2 = 1; y2 = 0; z2 = 0;
x3 = 0; y3 = 1; z3 = 0;
x4 = 0; y4 = 0; z4 = 1;
*/

// Points
?P1 = EV(x1, y1, z1);
?P2 = EV(x2, y2, z2);
?P3 = EV(x3, y3, z3);
?P4 = EV(x4, y4, z4);

// Join of 4 points → 3-blade
?Join4 = P1 ^ P2 ^ P3 ^ P4;

// Visualization
:Blue;
:P1;

:Red;
:P2;

:Yellow;
:P3;

:Cyan;
:P4;

:Green;
:Join4;
